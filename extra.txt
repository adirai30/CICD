name: Flask CI
on:
  push:
    branches: [ "master" ]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v3

    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.10'

    - name: Install dependencies
      run: |
        pip install -r requirements.txt

    - name: Run Flask Test
      run: |
        python -m py_compile app.py
  
    - name: Docker Build 
      run: docker build -t shahil004/url-shortener2 .
